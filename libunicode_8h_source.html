68 static inline int cr_add_point(CharRange *cr, uint32_t v)
69 {
70  if (cr->len >= cr->size) {
71  if (cr_realloc(cr, cr->len + 1))
72  return -1;
73  }
74  cr->points[cr->len++] = v;
75  return 0;
76 }
