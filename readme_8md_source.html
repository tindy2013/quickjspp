1 QuickJSPP is QuickJS wrapper for C++. It allows you to easily embed Javascript engine into your program.
2 
3 QuickJS is a small and embeddable Javascript engine. It supports the ES2020 specification including modules, asynchronous generators and proxies. More info: <https://bellard.org/quickjs/>
4 
5 # Example
6 ```cpp
7 #include "quickjspp.hpp"
8 #include <iostream>
9 
10 class MyClass
11 {
12 public:
13  MyClass() {}
14  MyClass(std::vector<int>) {}
15 
16  double member_variable = 5.5;
17  std::string member_function(const std::string& s) { return "Hello, " + s; }
18 };
19 
20 void println(const std::string& str) { std::cout << str << std::endl; }
21 
22 int main()
23 {
24  qjs::Runtime runtime;
25  qjs::Context context(runtime);
26  try
27  {
28  // export classes as a module
29  auto& module = context.addModule("MyModule");
30  module.function<&println>("println");
31  module.class_<MyClass>("MyClass")
32  .constructor<>()
33  .constructor<std::vector<int>>("MyClassA")
34  .fun<&MyClass::member_variable>("member_variable")
35  .fun<&MyClass::member_function>("member_function");
36  // import module
37  context.eval("import * as my from 'MyModule'; globalThis.my = my;", "<import>", JS_EVAL_TYPE_MODULE);
38  // evaluate js code
39  context.eval("let v1 = new my.MyClass();" "\n"
40  "v1.member_variable = 1;" "\n"
41  "let v2 = new my.MyClassA([1,2,3]);" "\n"
42  "function my_callback(str) {" "\n"
43  " my.println(v2.member_function(str));" "\n"
44  "}" "\n"
45  );
46 
47  // callback
48  auto cb = (std::function<void(const std::string&)>) context.eval("my_callback");
49  cb("world");
50  }
51  catch(qjs::exception)
52  {
53  auto exc = context.getException();
54  std::cerr << (std::string) exc << std::endl;
55  if((bool) exc["stack"])
56  std::cerr << (std::string) exc["stack"] << std::endl;
57  return 1;
58  }
59 }
60 ```

62 # Installation
63 QuickJSPP is header-only - put quickjspp.hpp into your include search path.
64 Compiler that supports C++17 or later is required.
65 The program needs to be linked against QuickJS.
66 Sample CMake project files are provided.
